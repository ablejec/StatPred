\section{Paket \pkg{mvtnorm}}

Paket \pkg{mvtnorm} vsebuje funkcije, s katerimi lahko obvladamo osnovne štiri naloge v zvezi z multivariatno normalno on multivariatno $t$ porazdelitvijo.

\begin{Schunk}
\begin{Sinput}
> library(mvtnorm)
\end{Sinput}
\end{Schunk}

\subsection{Bivariatna normalna porazdelitev}

\begin{Schunk}
\begin{Sinput}
> n <- 2
> dx <- 0.2
> mu <- rep(0, n)
> sigma <- diag(n)
> sigma <- matrix(c(1, -0.3, -0.3, 4), ncol = n)
> x1 <- seq(-5, 5, dx)
> x2 <- seq(-5, 5, dx)
> x <- expand.grid(x1 = x1, x2 = x2)
> z <- matrix(dmvnorm(x, mu, sigma), nrow = length(x1), 
+     ncol = length(x2))
> for (i in seq(-30, 30, 5)) persp(x1, x2, z, phi = 30, 
+     theta = i)
\end{Sinput}
\end{Schunk}
